config firewall address
    edit "Branch-NET"
        set subnet 10.1.0.0 255.255.0.0
    next
	edit "Datacenter"
        set subnet 192.168.1.0 255.255.255.0
    next
end

config system sdwan
    set status enable
    config zone
        edit "virtual-wan-link"
        next
        edit "SASE"
        next
        edit "WAN1"
        next
        edit "WAN2"
        next
		edit "HUB1"
		next
    end
    config members
    edit 1
            set interface "port1"
            set zone "WAN1"
            set comment "WAN1"
        next
    edit 2
            set interface "port2"
            set zone "WAN2"
            set cost 10
            set comment "WAN2"
        next
	edit 3
			set interface "HUB1-VPN1"
            set zone "HUB1"
            set comment "Mapping to HUB1 through WAN1"
        next
        edit 4
            set interface "HUB1-VPN2"
            set zone "HUB1"
            set cost 10
            set comment "Mapping to HUB1 through WAN2. Cost 10 since less preferred"
        next
    end
    config health-check
    edit "Internet"
            set server "1.1.1.1"
            set failtime 3
            set recoverytime 3
            set update-static-route disable
            set members 1 2
            config sla
                edit 1
                    set latency-threshold 250
                    set jitter-threshold 55
                    set packetloss-threshold 1
                next
            end
        next
	edit "HUB1_HC"
            set server "172.16.100.1"
            set failtime 3
            set update-static-route disable
            set members 3 4
            config sla
                edit 1
                    set latency-threshold 125
                    set jitter-threshold 55
                    set packetloss-threshold 1
                next
            end
        next
    end
    config service
		edit 2
			set name "Branch_to_DC1"
            set mode sla
            set dst "Datacenter" 
            set src "Branch-NET"
            config sla
                edit "HUB1_HC"
                    set id 1 
            end
            set priority-members 3 4
        next
        edit 1
            set name "WAN1-Primary_WAN2-Backup"
            set mode sla
            set dst "all"
            set src "Branch-NET"
            config sla
                edit "Internet"
                    set id 1
                next
            end
            set priority-members 1 2
        next
    end
end


config firewall policy
edit 1
        set name "Local Internet Breakout"
        set srcintf "port3"
        set dstintf "WAN1" "WAN2"
        set action accept
        set srcaddr "Branch-NET"
        set dstaddr "all"
        set schedule "always"
        set service "ALL"
        set logtraffic all
        set nat enable
    next
edit 2
        set name "Branch to Datacenter"
        set srcintf "port3"
        set dstintf "HUB1"
        set action accept
        set srcaddr "Branch-NET"
        set dstaddr "Datacenter"
        set schedule "always"
        set service "ALL"
        set logtraffic all
    next
end
